\subsection{Semantics of sequential expressions}
\begin{prooftree}
	\LeftLabel{(Sleep1)}
	\AxiomC{$n \geq 0$}
	\UnaryInfC{$\theta, timer:sleep(n+1),\mathcal{S}  \xrightarrow{\delta } \theta, timer:sleep(n),\mathcal{S} $}
  \end{prooftree}

\begin{prooftree}
	\LeftLabel{(Sleep2)}
	\AxiomC{}
	\UnaryInfC{$\theta, timer:sleep(0),\mathcal{S}  \xrightarrow{\tau } \theta,0,\mathcal{S} $}
  \end{prooftree}

\begin{prooftree}
	\LeftLabel{(SleepSeq)}
	\AxiomC{$\delta(<\theta,\mathcal{C}[e],\mathcal{S}>) $}
	\AxiomC{$\theta,e,\mathcal{S}\xrightarrow{\delta }\theta,e',\mathcal{S}$}
	\BinaryInfC{$\theta,\mathcal{C}[e],\mathcal{S}\xrightarrow{\delta }\theta,\mathcal{C}[e'],\mathcal{S}$}
  \end{prooftree}
