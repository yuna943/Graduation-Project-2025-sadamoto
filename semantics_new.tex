$\hspace{24mm}\wedge\hspace{1mm}\theta,e,\mathcal{S}\overset{send}{\not\rightarrow},$

\subsection{Semantics of sequential expressions}

	\LeftLabel{(receive+)}

  \end{prooftree}

\subsection{Forward reversible semantics}
\begin{prooftree}
	\LeftLabel{(receive1)}
	\AxiomC{
		$\begin{aligned}
		& \delta(\Pi)\wedge\hspace{1mm}Time(cl^{To}) = 0\hspace{1mm}\wedge  \\
		& \theta,e,\mathcal{S}\xrightarrow{rec(\kappa,\overline{cl},cl^{To})}\theta',e',\mathcal{S}'\hspace{1mm}\wedge \underset{p_i\in\Pi}{\cup }matchrec({\overline{cl_n}}^i\theta,v)=\varnothing 
		\end{aligned}$
		}
\UnaryInfC{$\Gamma;<p,\theta,e,\mathcal{S}>|\Pi\hspace{1mm}\hookrightarrow \hspace{1mm}\Gamma;<p,(\theta,e'\{\kappa\mapsto e^{To}\}),\mathcal{S}'>|\Pi$}
\end{prooftree}

\begin{prooftree}
	\LeftLabel{(receive2)}
	\AxiomC{

		& \forall i\Bigg[\delta(<\theta,e,\mathcal{S}>)\wedge\hspace{1mm}Time(cl_i^{To}) > 0\hspace{1mm}\wedge  \\
